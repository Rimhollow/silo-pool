Overview

A wrapper for the data returned by the PoolFixity class that can be used in a space-efficient rack response. It returns an XML document with the fixity data provided by a PoolFixity object, a piece at a time.

def each

  yield '<fixities hostname="'  + StoreUtils.xml_escape(@hostname)             + '" ' +
             'stored_before="'  + StoreUtils.xml_escape(@header.stored_before) + '" ' +
        'fixity_check_count="'  + @header.count.to_s                           + '" ' +
     'earliest_fixity_check="'  + @header.earliest.to_s                        + '" ' +
       'latest_fixity_check="'  + @header.latest.to_s                          + '">' + "\n"

  @list.each do |fix|
    yield '  <fixity name="'   + StoreUtils.xml_escape(fix.name)     + '" '  +
                'location="'   + StoreUtils.xml_escape(fix.location) + '" '  +
                    'sha1="'   + fix.sha1                            + '" '  +
                     'md5="'   + fix.md5                             + '" '  +
                    'size="'   + fix.size.to_s                       + '" '  +
             'fixity_time="'   + fix.fixity_time                     + '" '  +
                'put_time="'   + fix.put_time                        + '" '  +
                  'status="'   + fix.status                          + '"/>' + "\n"
  end

  yield "</fixities>\n"
end
